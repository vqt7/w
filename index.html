#include #include // include time.h for measuring execution time #define MAX_SIZE 100 void warshall(int n, int graph[MAX_SIZE][MAX_SIZE]) { int i, j, k; for (k = 0; k < n; k++) { for (i = 0; i < n; i++) { for (j = 0; j < n; j++) { graph[i][j] = graph[i][j] || (graph[i][k] && graph[k][j]); } } } } int main() { int n, i, j; int graph[MAX_SIZE][MAX_SIZE]; printf("Enter the number of nodes in the graph: "); scanf("%d", &n); printf("Enter the adjacency matrix of the graph:\n"); for (i = 0; i < n; i++) { for (j = 0; j < n; j++) { scanf("%d", &graph[i][j]); } } clock_t start_time = clock(); // Measure start time warshall(n, graph); clock_t end_time = clock(); // Measure end time double time_taken = ((double)(end_time - start_time)) / CLOCKS_PER_SEC; printf("Time taken for Warshall's algorithm: %f seconds\n", time_taken); // Space Complexity: O(n^2) for the adjacency matrix printf("Space Complexity: O(n^2)\n"); printf("Transitive closure of the graph:\n"); for (i = 0; i < n; i++) { for (j = 0; j < n; j++) { printf("%d ", graph[i][j]); } printf("\n"); } return 0; }
